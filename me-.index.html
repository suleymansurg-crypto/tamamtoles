        hum.Health = 0
    end

🏆 Leaderboard Luau

-- Leaderboard by سليمان
local Players = game:GetService("Players")
Players.PlayerAdded:Connect(function(p)
    local ls = Instance.new("Folder")
    ls.Name = "leaderstats"
    ls.Parent = p
    local pts = Instance.new("IntValue")
    pts.Name = "Points"
    pts.Value = 0
    pts.Parent = ls
end)
